mission "bunrodea defense (job)"
	job
	repeat
	name "Destroy a Korath ship"
	description "A Korath ship entered Bunrodea territory at Era Natta. Destroy it."
	source 
		government "Bunrodea" "Bunrodea (Guard)"
	waypoint "Era Natta"
	to offer
		has "bunrodea defense: done"
	npc kill
		government "Korath"
		personality personality heroic nemesis staying
		system "Era Natta"
		fleet
			names "korath"
			cargo 200
			variant 2
				"Palavret"
				"'olofez" 2
			variant 1
				"Rano'erek"
				"'olofez" 2
		dialog "You have destroyed the Korath ship."
	on complete
		dialog `You receive a message from the Bunrodea authorities, gratulating for your victory over the Korath ship.`
		"reputation: Bunrodea" += 10
		"reputation: Bunrodea (Erabu)" += 10
		"reputation: Bunrodea (Guard)" += 10
		"reputation: Bunrodea (Megasa)" += 10

mission "cargo transport small"
	job
	repeat
	name "Delivery to <planet>"
	description "Deliver <cargo> to <destination>. Payment is <payment>."
	cargo random 10 100
	to offer
		has "bunrodea defense: done"
		random < 75
	source
		government "Bunrodea" "Bunrodea (Guard)"
	destination
		distance 1 7
		government "Bunrodea" "Bunrodea (Guard)"
	on visit
		dialog phrase "generic cargo on visit"

		payment
